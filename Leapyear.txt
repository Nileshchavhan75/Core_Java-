import java.util.*;
public class Leapyear{
public static void main(String args[]){
Scanner sc = new Scanner(System.in);
int year = sc.nextInt();
if(year % 4 == 0; && year % 100 != 0;){
System.out.print(year +"this is leap Year");
}
else if(year % 400 = 0){
System.out.print(year + "this is leap year");
}else{
System.out.print(year+ "it is not leap year"); 
}
}
}
